## Implementation

### Contract Location

`OpenCapTable-v27/daml/Fairmint/OpenCapTable/CouponMinter.daml`

### Template

```daml
module Fairmint.OpenCapTable.CouponMinter where

import DA.List (replicateA)
import Splice.Api.FeaturedAppRightV1 (FeaturedAppRight(..), FeaturedAppRight_CreateActivityMarker(..), AppRewardBeneficiary(..), FeaturedAppActivityMarker)

data MintCouponsResult = MintCouponsResult with
    markerCids: [ContractId FeaturedAppActivityMarker]
  deriving (Eq, Show)

template CouponMinter
  with
    operator: Party
  where
    signatory operator

    nonconsuming choice MintCoupons : MintCouponsResult
      with
        featuredAppRight: ContractId FeaturedAppRight
        count: Int
        beneficiaries: [AppRewardBeneficiary]
        metadata: Optional Text
      controller operator
      do
        assertMsg "count must be at least 1" (count >= 1)

        allMarkerCids <- replicateA count $ do
          result <- exercise featuredAppRight FeaturedAppRight_CreateActivityMarker with
            beneficiaries
          pure result.activityMarkerCids

        pure MintCouponsResult with
          markerCids = concat allMarkerCids
```
